Unlocks the whole research tree on load

Unlocking part of the game with the mod was done like this
    1. Run the mod when the game is stared
    2. Loop through all the recipes
    3. Unlock each recipe in the game session
export default async () => {
	//Trigger on every new session incase enabled mid-game
	game.ee.on("game_started", () => {
		let recipes = Object.keys(_BLUEPRINTS.RECIPES); //  Get all the recipe names in the game
		for (let i = 0; i < recipes.length; i++) { //Loop through each recipe
			game.session.unlockRecipe(recipes[i]); // Unlock the recipe with this name
		}
		game.notify("All recipes unlocked"); // Tell the player they have unlocked everything
	});
};
